## 1. 系统总体架构

### 1.1 核心设计理念

Nexus Core 的核心设计理念是实现 **"从被动响应到主动服务"** 的范式转变：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                    【设计理念对比】                                       │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                         │
│   传统AI助手（被动模式）              Nexus Core（主动模式）              │
│   ┌─────────────────────┐            ┌─────────────────────┐           │
│   │  用户: "帮我开灯"    │            │  [无需唤醒]          │           │
│   │         ↓            │            │  系统主动感知环境    │           │
│   │  系统: "好的，开灯"  │            │         ↓            │           │
│   │         ↓            │            │  检测到光线不足      │           │
│   │  执行: 开灯          │            │         ↓            │           │
│   └─────────────────────┘            │  预判用户需要照明    │           │
│                                       │         ↓            │           │
│   特点: 等待→响应→执行               │  主动询问或直接开灯  │           │
│                                       └─────────────────────┘           │
│                                                                         │
│                                       特点: 感知→预判→主动服务          │
└─────────────────────────────────────────────────────────────────────────┘
```

### 2.2 核心接口定义

```python
# ==================== 核心接口契约定义 ====================
from abc import ABC, abstractmethod
from typing import Protocol, List, Dict, Any, Optional
from dataclasses import dataclass, field
from enum import Enum
import uuid
from datetime import datetime

# -------------------- 基础数据类型 --------------------

class UserGroup(Enum):
    """用户群体分类"""
    ELDERLY = "elderly"           # 老年人
    ADULT = "adult"               # 中青年
    YOUTH = "youth"               # 青少年
    CHILD = "child"               # 儿童
    SPECIAL_NEEDS = "special"     # 特殊需求群体

class ServiceMode(Enum):
    """服务模式"""
    PASSIVE = "passive"           # 被动响应
    PROACTIVE = "proactive"       # 主动服务
    COMPANION = "companion"       # 陪伴模式
    EMERGENCY = "emergency"       # 紧急模式

@dataclass
class UserProfile:
    """用户画像 - 千人千面的核心"""
    user_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    
    # 基础信息
    name: str = ""
    age: int = 0
    group: UserGroup = UserGroup.ADULT
    
    # 健康信息
    health_conditions: List[str] = field(default_factory=list)  # 健康状况
    medications: List[Dict] = field(default_factory=list)       # 用药信息
    mobility_level: float = 1.0                                 # 行动能力 0-1
    
    # 偏好设置
    communication_style: str = "gentle"      # 沟通风格: gentle/direct/formal
    volume_preference: float = 0.7           # 音量偏好 0-1
    proactive_level: float = 0.5             # 主动服务级别 0-1
    privacy_level: float = 0.5               # 隐私保护级别 0-1
    
    # 行为模式（通过学习获得）
    daily_patterns: Dict[str, Any] = field(default_factory=dict)   # 日常作息
    activity_patterns: Dict[str, Any] = field(default_factory=dict) # 活动习惯
    social_patterns: Dict[str, Any] = field(default_factory=dict)   # 社交习惯
    
    # 情感状态追踪
    emotional_baseline: Dict[str, float] = field(default_factory=dict)
    current_mood: str = "neutral"
    
    # 学习元数据
    profile_version: int = 1
    last_updated: datetime = field(default_factory=datetime.now)
    learning_samples: int = 0

@dataclass
class ContextFrame:
    """上下文帧 - 用于提示词注入"""
    timestamp: datetime
    
    # 环境上下文
    location: str = ""                       # 当前位置
    scene: str = ""                          # 场景类型
    time_of_day: str = ""                    # 时段
    weather: str = ""                        # 天气
    
    # 用户状态
    user_activity: str = ""                  # 当前活动
    user_posture: str = ""                   # 姿态
    user_emotion: str = ""                   # 情绪
    people_present: List[str] = field(default_factory=list)  # 在场人员
    
    # 设备状态
    available_devices: List[str] = field(default_factory=list)
    device_states: Dict[str, Any] = field(default_factory=dict)
    
    # 历史上下文
    recent_events: List[str] = field(default_factory=list)
    pending_tasks: List[str] = field(default_factory=list)

# -------------------- 模块接口协议 --------------------

class IPerceptionModule(Protocol):
    """感知模块接口"""
    def capture_frame(self) -> 'PerceptionFrame': ...
    def start_continuous_capture(self, fps: int = 10) -> None: ...
    def stop_capture(self) -> None: ...

class IAnalysisModule(Protocol):
    """分析模块接口"""
    def analyze(self, frame: 'PerceptionFrame', context: ContextFrame) -> 'AnalysisResult': ...
    def identify_person(self, face_embedding: Any) -> Optional['PersonIdentity']: ...
    def detect_anomaly(self, frame: 'PerceptionFrame') -> List['Anomaly']: ...

class IDecisionModule(Protocol):
    """决策模块接口"""
    def recognize_needs(self, analysis: 'AnalysisResult', profile: UserProfile) -> List['Need']: ...
    def plan_task(self, need: 'Need', context: ContextFrame) -> 'TaskPlan': ...
    def should_intervene(self, need: 'Need', profile: UserProfile) -> bool: ...

class IPersonalizationEngine(Protocol):
    """个性化引擎接口"""
    def get_profile(self, user_id: str) -> UserProfile: ...
    def update_profile(self, user_id: str, observation: Dict) -> None: ...
    def adapt_response(self, response: str, profile: UserProfile) -> str: ...
    def predict_preference(self, user_id: str, context: ContextFrame) -> Dict: ...

class IPromptEngine(Protocol):
    """提示词引擎接口"""
    def build_system_prompt(self, profile: UserProfile, context: ContextFrame) -> str: ...
    def inject_context(self, base_prompt: str, context: ContextFrame) -> str: ...
    def build_cot_prompt(self, task: str, examples: List[Dict]) -> str: ...

class ILearningModule(Protocol):
    """持续学习模块接口"""
    def record_interaction(self, interaction: 'Interaction') -> None: ...
    def update_model(self, feedback: 'Feedback') -> None: ...
    def consolidate_memory(self) -> None: ...
```

                "tech_savvy": True,

    def create_profile(self, user_id: str, initial_data: Dict) -> UserProfile:

        age = initial_data.get("age", 30)
        if age >= 60:
            group = UserGroup.ELDERLY
        elif age >= 18:
            group = UserGroup.ADULT
        else:
            group = UserGroup.YOUTH
        
        # 2. 加载群体模板作为默认值
        template = self.group_templates.get(group, {})
        
        # 3. 创建画像
        profile = UserProfile(
            user_id=user_id,
            name=initial_data.get("name", ""),
            age=age,
            group=group,
            communication_style=template.get("communication_style", "gentle"),
            volume_preference=template.get("volume_preference", 0.7),
            proactive_level=template.get("proactive_level", 0.5),
        )
        
        self.profiles[user_id] = profile
        return profile
    
    def update_from_observation(self, user_id: str, observation: Dict) -> None:
        """根据观察数据更新画像"""
        profile = self.profiles.get(user_id)
        if not profile:
            return
        
        # 更新日常模式
        if "activity" in observation:
            self._update_activity_pattern(profile, observation)
        
        # 更新情感基线
        if "emotion" in observation:
            self._update_emotional_baseline(profile, observation)
        
        # 更新偏好
        if "feedback" in observation:
            self._update_preferences(profile, observation)
        
        profile.learning_samples += 1
        profile.last_updated = datetime.now()
    
    def _update_activity_pattern(self, profile: UserProfile, obs: Dict) -> None:
        """更新活动模式"""
        time_slot = obs.get("time_slot")  # 例如 "07:00-08:00"
        activity = obs.get("activity")    # 例如 "breakfast"
        
        if time_slot not in profile.daily_patterns:
            profile.daily_patterns[time_slot] = {}
        
        patterns = profile.daily_patterns[time_slot]
        if activity not in patterns:
            patterns[activity] = 0
        patterns[activity] += 1
    
    def _update_emotional_baseline(self, profile: UserProfile, obs: Dict) -> None:
        """更新情感基线（指数移动平均）"""
        alpha = 0.1  # 学习率
        emotion = obs.get("emotion", "neutral")
        
        if emotion not in profile.emotional_baseline:
            profile.emotional_baseline[emotion] = 0.0
        
        # 指数移动平均更新
        profile.emotional_baseline[emotion] = (
            alpha * 1.0 + (1 - alpha) * profile.emotional_baseline[emotion]
        )
    
    def get_personalized_config(self, user_id: str) -> Dict:
        """获取个性化配置（用于提示词注入）"""
        profile = self.profiles.get(user_id)
        if not profile:
            return {}
        
        return {
            "user_name": profile.name,
            "user_group": profile.group.value,
            "communication_style": profile.communication_style,
            "proactive_level": profile.proactive_level,
            "health_conditions": profile.health_conditions,
            "current_mood": profile.current_mood,
            "daily_patterns": profile.daily_patterns,
            "preferences": {
                "volume": profile.volume_preference,
                "privacy": profile.privacy_level,
            }

class BehaviorPredictor:
    """行为预测器 - 预测用户下一步可能的需求"""
    
    def __init__(self, profile_manager: UserProfileManager):
        self.profile_manager = profile_manager
    
    def predict_next_need(self, user_id: str, context: ContextFrame) -> List[Dict]:
        """预测用户接下来可能的需求"""
        profile = self.profile_manager.profiles.get(user_id)
        if not profile:
            return []
        
        predictions = []
        current_time = context.timestamp.strftime("%H:%M")
        
        # 1. 基于时间模式预测
        time_based = self._predict_from_time_pattern(profile, current_time)
        predictions.extend(time_based)
        
        # 2. 基于当前活动预测
        activity_based = self._predict_from_activity(profile, context.user_activity)
        predictions.extend(activity_based)
        
        # 3. 基于情绪状态预测
        emotion_based = self._predict_from_emotion(profile, context.user_emotion)
        predictions.extend(emotion_based)
        
        # 4. 按置信度排序
        predictions.sort(key=lambda x: x["confidence"], reverse=True)
        
        return predictions[:5]  # 返回top5预测
    
    def _predict_from_time_pattern(self, profile: UserProfile, current_time: str) -> List[Dict]:
        """基于历史时间模式预测"""
        predictions = []
        
        # 查找当前时间段的历史活动
        for time_slot, activities in profile.daily_patterns.items():
            if self._time_in_slot(current_time, time_slot):
                total = sum(activities.values())
                for activity, count in activities.items():
                    confidence = count / total if total > 0 else 0
                    if confidence > 0.3:  # 置信度阈值
                        predictions.append({
                            "need_type": activity,
                            "confidence": confidence,
                            "source": "time_pattern",
                            "reason": f"用户通常在{time_slot}进行{activity}"
                        })
        
        return predictions
    
    def _time_in_slot(self, current: str, slot: str) -> bool:
        """判断当前时间是否在时间段内"""
        try:
            start, end = slot.split("-")
            return start <= current <= end
        except:
            return False

### 5.2 置信度加权融合算法

```python
class MultimodalFusion:
    """多模态置信度加权融合"""
    
    def fuse(self, vision_data: Dict, audio_data: Dict, 
             sensor_data: Dict, device_data: Dict) -> Dict:
        """融合多模态数据"""
        
        # 获取各模态置信度
        confidences = {
            "vision": vision_data.get("confidence", 0.5),
            "audio": audio_data.get("confidence", 0.5),
            "sensor": sensor_data.get("confidence", 0.8),
            "device": device_data.get("confidence", 0.9)
        }
        
        # 归一化权重
        total = sum(confidences.values())
        weights = {k: v/total for k, v in confidences.items()}
        
        # 加权融合场景判断
        scene_votes = {}
        for modality, data in [("vision", vision_data), ("audio", audio_data)]:
            scene = data.get("scene_type", "unknown")
            if scene not in scene_votes:
                scene_votes[scene] = 0
            scene_votes[scene] += weights[modality]
        
        # 最终场景判断
        final_scene = max(scene_votes, key=scene_votes.get)
        
        return {
            "scene_type": final_scene,
            "scene_confidence": scene_votes[final_scene],
            "user_activity": self._fuse_activity(vision_data, audio_data, weights),
            "emotion": self._fuse_emotion(vision_data, audio_data, weights),
            "fusion_weights": weights
        }

class OnlineLearner:
    """在线学习器 - 让系统不断理解用户"""
    
    def __init__(self, profile_manager: UserProfileManager):
        self.profile_manager = profile_manager
        self.learning_buffer = []  # 学习缓冲区
        self.update_threshold = 10  # 累积10条后触发更新
    
    def observe(self, user_id: str, observation: Dict) -> None:
        """记录观察数据"""
        self.learning_buffer.append({
            "user_id": user_id,
            "observation": observation,
            "timestamp": datetime.now()
        })
        
        # 达到阈值时触发增量更新
        if len(self.learning_buffer) >= self.update_threshold:
            self.incremental_update()
    
    def incremental_update(self) -> None:
        """增量更新用户画像"""
        for item in self.learning_buffer:
            self.profile_manager.update_from_observation(
                item["user_id"], 
                item["observation"]
            )
        self.learning_buffer.clear()
    
    def process_feedback(self, user_id: str, feedback: Dict) -> None:
        """处理用户反馈（显式学习）"""
        feedback_type = feedback.get("type")
        
        if feedback_type == "positive":  # 用户表示满意
            self._reinforce_behavior(user_id, feedback)
        elif feedback_type == "negative":  # 用户表示不满
            self._adjust_behavior(user_id, feedback)
        elif feedback_type == "correction":  # 用户纠正
            self._learn_correction(user_id, feedback)

class InterventionDecider:
    """干预决策器 - 决定是否主动介入"""
    
    def decide(self, need: Need, profile: UserProfile, context: ContextFrame) -> bool:
        """决定是否应该主动介入"""
        
        # 因素 1: 需求紧急程度
        urgency_score = need.urgency
        
        # 因素 2: 用户偏好的主动服务级别
        proactive_preference = profile.proactive_level
        
        # 因素 3: 场景适合度 (是否打扰)
        disturbance_risk = self._calculate_disturbance_risk(context)
        
        # 因素 4: 历史反馈 (用户对类似介入的反应)
        historical_acceptance = self._get_historical_acceptance(profile, need.need_type)
        
        # 综合评分
        score = (
            urgency_score * 0.4 +
            proactive_preference * 0.3 +
            (1 - disturbance_risk) * 0.2 +
            historical_acceptance * 0.1
        )
        
        # 阈值判断
        return score > 0.5

                voice_sim = cosine_similarity(voice_embedding, person.voice_embedding)
            
            # 综合评分
            score = face_sim * 0.7 + voice_sim * 0.3
            
            if score > best_score and score > 0.8:  # 阈值
                best_score = score
                best_match = person

class TaskOrchestrator:
    """任务编排器 - 协调多设备完成复杂任务"""
    
    def __init__(self, device_manager: DeviceManager):
        self.device_manager = device_manager
        self.conflict_resolver = ConflictResolver()
    
    def orchestrate(self, task: TaskPlan) -> ExecutionPlan:
        """编排任务执行"""
        
        # 1. 分解任务为子任务
        subtasks = self._decompose_task(task)
        
        # 2. 为每个子任务分配设备
        assignments = []
        for subtask in subtasks:
            device = self._find_best_device(subtask)
            assignments.append((subtask, device))
        
        # 3. 解决设备冲突
        resolved = self.conflict_resolver.resolve(assignments)
        
        # 4. 生成执行计划
        execution_plan = self._build_execution_plan(resolved)
        
        return execution_plan
    
    def _find_best_device(self, subtask: SubTask) -> RMADevice:
        """为子任务找到最合适的设备"""
        available = self.device_manager.get_available_devices()
        
        candidates = []
        for device in available:
            if subtask.required_capability in device.capabilities:
                score = self._score_device(device, subtask)
                candidates.append((device, score))
        
        if not candidates:
            raise NoDeviceAvailableError(subtask)
        
        return max(candidates, key=lambda x: x[1])[0]
```

class ConflictResolver:
    """冲突仲裁器"""
    
    def resolve(self, assignments: List[Tuple[SubTask, RMADevice]]) -> List:
        """解决设备冲突"""
        
        # 检测冲突
        conflicts = self._detect_conflicts(assignments)
        
        for conflict in conflicts:
            strategy = self._choose_strategy(conflict)
            
            if strategy == "preempt":  # 高优先级抢占
                self._preempt(conflict)
            elif strategy == "queue":  # 排队等待
                self._queue(conflict)
            elif strategy == "degrade":  # 降级执行
                self._degrade(conflict)
        
        return assignments
